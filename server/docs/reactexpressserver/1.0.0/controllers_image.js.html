const ImageStore = require("../models/Image");

/**
 * Upload a Multipart-File then saving it to MySQL database
 * @param {Object} req - request
 * @param {Object} res - response
 * @return {Object}  returns created Image
 */
exports.insert = function(req, res) {
  console.log(req.file);
  ImageStore.create({
    name: req.file.originalname,
    image: fs.readFileSync(`./uploads/${req.file.filename}`)
  }).then(result => {
    console.log(result);
    res.status(201).json({
      createdImage: {
        image: result
      }
    });

};

/**
 * Get all images from database
 * @param {Object} _req - request
 * @param {Object} res - response
 * @return {Object} - returns all Images
 */
exports.findAll = (_req, res) => {
  ImageStore.findAll()
    .then(results => {
      console.log(results);
      return res.json({
        data: results
      });
    })
    .catch(err => console.log(err));
};
